    function genTable(){
        var j=<table border='2'>
               
                <thead>
                    
                    <th>First Name</th>

                    <th>Age</th>
                    <th>Gender</th>

        function deleteRow(i){
            ar.splice(i,1)
            genTable()
            //console.log(ar)
        }
        function addEmp(e){
            e.preventDefault()
            ob.firstname=document.empForm.fname.value;
            ob.lastname=document.empForm.lname.value;
            ob.age=document.empForm.age.value;
            ob.gender=document.empForm.gen.value;
            console.log(ob)
            ar.push(ob);
            genTable()
            document.empForm.reset()
            ob={}
        }
        function EditRow(i){
            console.log('edit '+i)
            document.empForm.fname.value=ar[i].firstname
            document.empForm.lname.value=ar[i].lastname
            document.empForm.age.value=ar[i].age
            var genOpt=document.empForm.gen;
            genOpt.value=ar[i].gender
            genOpt.forEach(e=>{

                if(ar[i].gender==e.defaultValue)
