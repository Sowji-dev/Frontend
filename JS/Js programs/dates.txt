            function clickedDate(){
                var dob=new Date(document.getElementById('date').value);
                var d=new Date();
               //console.log(dob,d)
                var ms=d-dob;
                //console.log(ms)
                const minute = 1000 * 60;
                const hour = minute * 60;
                const day = hour * 24;
                const year = day * 365;
                let years = ms/ year;  
                console.log(`Age is ${years} years`)            
            }

        <button onclick="clickIn()" id="in">IN</button>
        <button onclick="clickOut()" id="out">OUT</button>
        <button onclick="clickTotal()" id="total">TOTAL</button>
        <script>
            document.getElementById('out').style.display='none'
            var indate,outDate,ar=[],s=0
            function clickIn(){
                  document.getElementById('in').style.display='none'  
                  document.getElementById('out').style.display='inline'
                  indate=new Date()
                 // console.log(indate)    
            }
            function clickOut(){
                document.getElementById('out').style.display='none'
                document.getElementById('in').style.display='inline'  
               outDate=new Date()    
               ar.push(outDate-indate)  
            }
            function clickTotal(){
                console.log(ar)
                ar.forEach(t=>{
                    s=s+t;
                })
                console.log(`${s} milliseconds`)
            }
 </script>
