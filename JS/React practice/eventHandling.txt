function App() {
  var [c,setC] = useState(0)
  function incr(){
    setC(c+1)
  }
  function decr(){
    setC(c-1)
  }
  function reset(){
    setC(0)
  }
  return (
    <div >
      <div >{c}</div>
      <button onClick={incr}>Increment</button>
      <button onClick={decr}>Decrement</button>
      <button onClick={reset}>Reset</button>
      </div>
  );
}
export default App;
